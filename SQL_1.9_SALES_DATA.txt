CREATE TABLE SALES_ORDER_DATA(
CHANNEL VARCHAR(50),
SALESORDERLINEKEY INT PRIMARY KEY,
SALESORDER VARCHAR(50),
SALESORDERLINE VARCHAR(50)
);

CREATE TABLE TERRITORY_DATA(
SALESTERRITORYKEY INT PRIMARY KEY,
REGION VARCHAR(50),
COUNTRY VARCHAR(50),
GROUPE VARCHAR(50)
);

CREATE TABLE SALES_DATA(
SALESORDERLINEKEY INT PRIMARY KEY,
RESELLERKEY INT,
CUSTOMERKEY INT,
PRODUCTKEY INT,
ORDERDATEKEY INT,
DUEDATEKEY INT,
SHIPDATEKEY INT,
SALESTERRITORYKEY INT,
ORDERQUANTITY INT,
UNITPRICE DECIMAL(5,2),
EXTENDEDAMT DECIMAL(5,2),
UNITPRICEDISTPCT DECIMAL(1,2),
PRODUCTSTNDCST DECIMAL(5,2),
TTLPRODUCTCST DECIMAL(5,2),
SALESAMT DECIMAL(5,2)
);

CREATE CLUSTERED INDEX IDX_SALESORDERLINEKEY
